class Solution {
public:
  bool containsDuplicate(vector<int>& nums) {
    if (nums.size() == 1) return false;

    unordered_map<int, int> myMap;
    for (int x: nums) {
      if (myMap[x]) return true;
      else myMap[x]++;
    }
    return false;
  }
};
